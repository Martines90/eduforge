### 4. Step Transitions
```typescript

  setCountry(country);  // Save to cookie

// Role â†’ Subject or Complete

  setIdentity(identity);  // Save to cookie
  if (identity === 'teacher') {
    setStep('subject');
  } else {
    completeOnboarding();
    setStep('complete');
  }
}

// Subject â†’ Action
handleSubjectSelect(subject) {
  setSubject(subject);  // Save to cookie
  setStep('action');
}

// Action â†’ Navigate & Complete

  completeOnboarding();
  setStep('complete');
  router.push(action === 'create' ? '/task_creator' : '/search_tasks');
}
```
